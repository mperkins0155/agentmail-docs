```typescript title="TypeScript"
import { AgentMailClient } from 'agentmail';

const client = new AgentMailClient();

// Define response templates
const templates = {
  greeting: 'Hello! Thanks for reaching out.',
  support: 'Our support team will get back to you within 24 hours.',
  closing: 'Best regards,\nThe Support Team',
};

// Auto-reply webhook handler
export async function handleIncomingEmail(event: any) {
  const { inboxId, messageId, subject, text } = event.data;
  
  // Analyze email content
  const sentiment = analyzeSentiment(text);
  
  // Generate appropriate response
  const response = generateResponse(subject, sentiment);

  });

```

```python title="Python"
from agentmail import AgentMail

client = AgentMail()

# Define response templates
templates = {
    'greeting': 'Hello! Thanks for reaching out.',
    'support': 'Our support team will get back to you within 24 hours.',
    'closing': 'Best regards,\nThe Support Team',
}

# Auto-reply webhook handler
def handle_incoming_email(event):
    inbox_id = event['data']['inboxId']
    message_id = event['data']['messageId']
    subject = event['data']['subject']
    text = event['data']['text']
    
    # Analyze email content
    sentiment = analyze_sentiment(text)
    
    # Generate appropriate response
    response = generate_response(subject, sentiment)
    
    # Send reply
    client.inboxes.messages.reply(
        inbox_id=inbox_id,
        message_id=message_id,
        text=response
    )
```
</CodeBlocks>

## AI-Powered Templates

### OpenAI + AgentMail

Build intelligent email agents with GPT models.

<Card title="OpenAI Integration" icon="fa-solid fa-brain" href="https://github.com/agentmail-to/template-openai">
  **Features:**
  - GPT-powered responses
  - Context-aware replies
  - Email summarization
  - Sentiment analysis
  - Intent classification
  - Multi-language support
</Card>

```typescript title="Example: GPT-Powered Auto-Reply"
import { AgentMailClient } from 'agentmail';
import OpenAI from 'openai';

const agentmail = new AgentMailClient();
const openai = new OpenAI();

export async function handleEmail(event: any) {
  const { inboxId, messageId, text } = event.data;

  // Generate intelligent response using GPT
  const completion = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [
      {
        role: 'system',
        content: 'You are a helpful customer support assistant.',
      },
      {
        role: 'user',
        content: `Reply to this email: ${text}`,
      },
    ],
  });

  const reply = completion.choices[0].message.content;

  // Send the AI-generated reply
  await agentmail.inboxes.messages.reply(inboxId, messageId, {
    text: reply,
  });
}
```

## Using Templates

### Quick Start

<Steps>
  <Step title="Choose a Template">
    Browse the template library and select one that matches your use case.
  </Step>

  <Step title="Clone or Fork">
    ```bash
    # Clone the template
    git clone https://github.com/agentmail-to/template-name.git my-project
    cd my-project
    
    # Or use GitHub CLI
    gh repo fork agentmail-to/template-name --clone
    ```
  </Step>

  <Step title="Install Dependencies">
    ```bash
    npm install
    # or
    pip install -r requirements.txt
    ```
  </Step>

  <Step title="Configure Environment">
    ```bash
    cp .env.example .env
    # Edit .env with your API keys
    ```
  </Step>

  <Step title="Customize">
    Modify the template to fit your specific requirements:
    - Update webhook handlers
    - Add custom logic
    - Integrate additional services
    - Adjust email templates
  </Step>

  <Step title="Test Locally">
    ```bash
    npm run dev
    # or
    python app.py
    ```
  </Step>

  <Step title="Deploy">
    Follow the deployment instructions in the template's README.
  </Step>
</Steps>

## Template Structure

All templates follow a consistent structure:

```
template-name/
├── README.md                 # Setup instructions and documentation
├── .env.example             # Environment variable template
├── .gitignore              # Git ignore rules
├── package.json            # Dependencies (Node.js)
├── requirements.txt        # Dependencies (Python)
├── tsconfig.json           # TypeScript configuration
├── src/                    # Source code
│   ├── index.ts           # Entry point
│   ├── handlers/          # Webhook handlers
│   ├── utils/             # Utility functions
│   └── types/             # Type definitions
├── tests/                  # Test files
├── docker/                 # Docker configuration
│   ├── Dockerfile
│   └── docker-compose.yml
├── .github/               # GitHub Actions workflows
│   └── workflows/
│       ├── ci.yml
│       └── deploy.yml
└── docs/                   # Additional documentation
```
